Step 1: Dataset Summary & Exploration¶

The pickled data is a dictionary with 4 key/value pairs:

  • 'features' is a 4D array containing raw pixel data of the traffic sign images, (num examples, width, height, channels).
  • 'labels' is a 1D array containing the label/class id of the traffic sign. The file signnames.csv contains id -> name mappings for each id.
  • 'sizes' is a list containing tuples, (width, height) representing the original width and height the image.
  • 'coords' is a list containing tuples, (x1, y1, x2, y2) representing coordinates of a bounding box around the sign in the image. THESE COORDINATES ASSUME THE ORIGINAL IMAGE. THE PICKLED DATA CONTAINS RESIZED VERSIONS (32 by 32) OF THESE IMAGES

# After we show the table with examples of each traffic sign type, we showed a histogram of how many examples
# exist for each type. We can see that there are very few examples for several types.
# So here, we will create variants of images from the input classes which don't have many actual examples. This will
# prevent the network from biasing too strongly towards the classes with more examples.
from scipy import ndimage
import random

# Similar data to what we have in num_examples_for_id, but this is a flat array instead of a dictionary.
bincount_examples = np.bincount(y_train)
#mean_examples_per_class = int(np.mean(bincount_examples)) #reasonable heuristic but can't run the cell multiple times safely
min_images_per_class = 500

# How much we want to rotate the images (absolute values - we'll actually do both positive and negative rotations).
min_degrees = 5 # inclusive
max_degrees = 20 # inclusive

# We want to make each class have at least the original Mean number of images.
for sign_id in num_examples_for_id:
    num_examples = num_examples_for_id[sign_id]
    
    # Generate rotated versions of original images
    indeces_of_pics_in_class = np.where(y_train == sign_id)[0]
    #print("indeces: ",indeces_of_pics_in_class)
    #print("Num in class: ",len(indeces_of_pics_in_class))
    #print("Num expected in class: ",num_examples)
    
    num_to_make = (min_images_per_class - num_examples)
    if(num_to_make > 0):
        images_to_add = []
        classifications_to_add = []
        
        for num in range(num_to_make):
            # Choose one of the images from this class.
            image_index = indeces_of_pics_in_class[random.randint(0, num_examples-1)]
            source_image = X_train[image_index]

            # Rotate the image & store it and its classification.
            rotated_image = ndimage.rotate(source_image,
                                            (random.choice([1, -1]) * random.randint(min_degrees, max_degrees)),
                                            axes=(1,0),
                                            reshape=False)
            images_to_add.append(rotated_image)
            classifications_to_add.append(sign_id)

        num_examples_for_id[sign_id] += len(images_to_add)

        # Need to append along a specific axis
        X_train = np.append(X_train, np.array(images_to_add), axis=0)
        y_train = np.append(y_train, np.array(classifications_to_add), axis=0)

# Grayscales all images in an array (4d array, 1st dim is all images, next 3 are the w/h/color-channels)
def grayscale(data):
    grayed_images = []
    for image in data:
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        grayed_images.append(gray)
    retVal = np.array(grayed_images)
    retVal = retVal[..., newaxis] # make it so it has the 4th dimension again (will be an array of length 1, holding just the grayscaled pixel value)
    return retVal

INPUT_DEPTH = NUM_CHANNELS # basically, how many color channels there are
conv2_layer = None # global variable so that we can feed this into the analyzer in the last step

def LeNet(x):
    global conv2_layer

    # Arguments used for tf.truncated_normal, randomly defines variables for the weights and biases for each layer
    mu = 0
    sigma = 0.1

    print("INPUT TENSOR: ",x)
    
    # Layer 1: Convolutional. Input = 32x32x1. Output = 28x28xOUTPUT_DEPTH.
    # OUTPUT HEIGHT = (input_height - filter_height + 1)/vertical_stride
    # https://www.tensorflow.org/versions/r0.11/api_docs/python/nn/convolution#conv2d
    # To cause valid-padding to have 2 pixels of lost overhang ON EACH SIDE per dimension, we need a filter of this size:
    filter_height = 5
    filter_width = 5
    output_depth1 = 16
    F_W = tf.Variable(tf.truncated_normal( [filter_height, filter_width, INPUT_DEPTH, output_depth1], mean=mu, stddev=sigma ))
    F_b = tf.Variable(tf.zeros( output_depth1 ))
    x = tf.nn.conv2d(x, F_W, strides=[1, 1, 1, 1], padding='VALID') + F_b
    print("AFTER CONV: ", x)

    # Activation.
    # https://www.tensorflow.org/versions/r0.11/api_docs/python/nn/activation_functions_#relu
    x = tf.nn.relu(x)
    #x = tf.nn.dropout(x, keep_prob)
    print("AFTER ACTIVATION: ",x)

    # Pooling. Input = 28x28xoutput_depth1. Output = 14x14xoutput_depth1.
    # https://www.tensorflow.org/versions/r0.11/api_docs/python/nn.html#max_pool
    #x = tf.nn.max_pool(x, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID')
    x = tf.nn.avg_pool(x, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID')
    print("AFTER POOLING: ",x)

    # Layer 2: Convolutional. Input=14x14xoutput_depth1 Output = 10x10xoutput_depth2.
    filter_height2 = 5
    filter_width2 = 5
    output_depth2 = 32
    F_W2 = tf.Variable(tf.truncated_normal( [filter_height2, filter_width2, output_depth1, output_depth2], mean=mu, stddev=sigma))  # the output of 1 is input-depth of layer 2, etc.
    F_b2 = tf.Variable(tf.zeros( output_depth2 ))
    # Set the stride for each dimension (batch_size, height, width, depth)
    strides2 = [1, 1, 1, 1]
    padding2 = 'VALID'
    # NOTE: We use "conv2" variable here instead of our usual X, because we want to pass this specific
    # layer into a visualizer at the bottom of this notebook.
    conv2_layer = tf.nn.conv2d(x, F_W2, strides2, padding2) + F_b2
    print("AFTER SECOND CONVOLUTION: ",conv2_layer)
    
    # Activation.
    x = tf.nn.relu(conv2_layer)
    #x = tf.nn.dropout(x, keep_prob)
    print("AFTER SECOND ACTIVATION: ",x)

    # Pooling. Input = 10x10xoutput_depth2. Output = 5x5xoutput_depth2.
    #x = tf.nn.max_pool(x, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID')
    x = tf.nn.avg_pool(x, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID')
    print("AFTER SECOND POOLING: ",x)

    # Flatten. Input = 5x5xoutput_depth2. Output = flattened array of size 5*5*output_depth2.
    x = tf.contrib.layers.flatten(x)
    print("AFTER FLATTENING: ",x)
    
    # Layer 3: Fully Connected. Input = 400. Output = 120.
    #l3_numInput = 5*5*output_depth2
    l3_numInput = int(x.get_shape()[1])
    l3_numOutput = 120
    F_W3 = tf.Variable(tf.random_normal( [l3_numInput, l3_numOutput], mu, sigma))
    F_b3 = tf.Variable(tf.zeros( l3_numOutput ))
    x = tf.add(tf.matmul(x, F_W3), F_b3)
    
    # Activation.

    x = tf.nn.dropout(x, keep_prob)
    print("AFTER THIRD ACTIVATION: ",x)

    # Layer 4: Fully Connected. Input = 120. Output = 84.
    l4_numOutput = 84
    F_W4 = tf.Variable(tf.random_normal([l3_numOutput, l4_numOutput], mu, sigma))
    F_b4 = tf.Variable(tf.zeros( l4_numOutput ))
    x = tf.add(tf.matmul(x, F_W4), F_b4)
    
    # Activation.
    x = tf.nn.relu(x)
    x = tf.nn.dropout(x, keep_prob)

def evaluate(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    sess = tf.get_default_session()

    for offset in range(0, num_examples, BATCH_SIZE):
        batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
        accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y, keep_prob: KEEP_ALL})
        total_accuracy += (accuracy * len(batch_x))
    return total_accuracy / num_examples

            sess.run(training_operation, feed_dict={x: batch_x, y: batch_y, keep_prob: KEEP_PROB})
